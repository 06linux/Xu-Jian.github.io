{{ site.time | date\_to\_xmlschema }}     # =\> 2008-11-07T13:07:54-08:00

{{ site.time | date\_to\_string }}        # =\> 07 Nov 2008
{{ site.time | date\_to\_long\_string }}   # =\> 07 November 2008
