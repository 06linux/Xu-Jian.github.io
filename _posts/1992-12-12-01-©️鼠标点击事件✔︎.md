	function eventBroker(e, className, fn) {    
	  var target = e.target;  
	  while(target) {   
	    if(target && target.nodeName == '#document') {   
	      break;    
	    } else if(target.classList.contains(className)) {  
	      fn();   
	      break;   
	    };   
	    target = target.parentNode;   
	  };   
	}
	

	addEvent(document, 'click', function(e){
	  eventBroker(e, 'test', function(){});
	});
