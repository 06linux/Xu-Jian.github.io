## 正方形

~~~
#square {
	width: 100px;
	height: 100px;
	background: red;
}
~~~
{: .language-ruby}





## 长方形

~~~
#rectangle {
	width: 200px;
	height: 100px;
	background: red;
}
~~~
{: .language-ruby}







## 平行四边形

~~~
#parallelogram {
	width: 150px;
	height: 100px;
	-webkit-transform: skew(20deg);
	  -moz-transform: skew(20deg);
	    -o-transform: skew(20deg);
	background: red;
}
~~~
{: .language-ruby}



## 梯形
~~~
#trapezoid {
	border-bottom: 100px solid red;
	border-left: 50px solid transparent;
	border-right: 50px solid transparent;
	height: 0;
	width: 100px;
}
~~~
{: .language-ruby}







## 菱形
~~~
#diamond {
    width: 0;
    height: 0;
    border: 50px solid transparent;
    border-bottom-color: red;
    position: relative;
    top: -50px;
}
#diamond:after {
    content: '';
    position: absolute;
    left: -50px;
    top: 50px;
    width: 0;
    height: 0;
    border: 50px solid transparent;
    border-top-color: red;
}
~~~
{: .language-ruby}
